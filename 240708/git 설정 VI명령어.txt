git 최초설정
- 사용자 이름과 이메일 주소 설정
git config --global user.name "영문이름"
git config --global use.email "이메일주소)

// git에서 작업하는 공간 branch ---> master
git config --global init.defaultBranch main

git branch -M main

2.git 형상관리 시작하기
git init -- 깃에 형상관리를 맡김
git status -- 깃의 정보를 확인
